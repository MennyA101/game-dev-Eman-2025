JavaScript Loops — One‑Page Demo

Interactive examples of common JavaScript loops. Click the buttons on the right to run each example and see the output.

How to use

Click any "Run" button to execute that loop. The code shown is the exact code that runs (browser console & this page).

1) for loop

// standard for loop
for (let i = 0; i < 5; i++) {
output.push(`for: ${i}`);
}

2) while loop

// while loop
let j = 0;
while (j < 5) {
output.push(`while: ${j}`);
j++;
}

3) do...while loop

// do..while - runs at least once
let k = 0;
do {
output.push(`do...while: ${k}`);
k++;
} while (k < 3);

4) for...in loop

// for...in iterates object keys
const obj = {a:1, b:2, c:3};
for (const key in obj) {
if (Object.hasOwn(obj, key)) output.push(`for...in: ${key} -> ${obj[key]}`);
}

5) for...of loop

// for...of iterates iterable values
const arr = ['a','b','c'];
for (const val of arr) {
output.push(`for...of: ${val}`);
}

6) Array forEach (callback)

// Array.prototype.forEach
[10,20,30].forEach((v, idx) => output.push(`forEach: idx=${idx} value=${v}`));

7) Labeled & nested loops (with break/continue)

outer: for (let a=1; a<=3; a++) {
for (let b=1; b<=3; b++) {
if (a * b === 4) break outer; // jump out of both loops
if (b === 2) continue; // skip this inner iteration
output.push(`labelled: ${a},${b}`);
}
}
